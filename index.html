Python Основы

Основные конструкции

# Условные операторы
if condition:
    # выполняется если condition истинно
    print("Условие истинно")
elif another_condition:
    # выполняется если another_condition истинно
    print("Другое условие истинно")
else:
    print("Все условия ложны")

# Циклы
for i in range(5):  # 0, 1, 2, 3, 4
    print(i)

while condition:
    # выполняет пока condition истинно
    print("Выполняюсь...")
    # не забудьте изменить condition чтобы избежать бесконечного цикла!
Совет

Используйте list comprehensions для создания списков в одну строку: [x**2 for x in range(10) if x % 2 == 0]

PyGame Основы

Минимальное приложение PyGame

import pygame
import sys

# Инициализация Pygame
pygame.init()

# Создание окна
screen = pygame.display.set_mode((800, 600))
pygame.display.set_caption("Моя игра")

# Цвета
WHITE = (255, 255, 255)
BLUE = (0, 120, 255)

# Главный цикл игры
running = True
while running:
    # Обработка событий
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
        elif event.type == pygame.KEYDOWN:
            if event.key == pygame.K_ESCAPE:
                running = False
    
    # Очистка экрана
    screen.fill(WHITE)
    
    # Рисование
    pygame.draw.rect(screen, BLUE, (350, 250, 100, 100))
    
    # Обновление экрана
    pygame.display.flip()
    
    # Ограничение FPS
    pygame.time.Clock().tick(60)

# Завершение работы
pygame.quit()
sys.exit()
Важно

Всегда вызывайте pygame.quit() перед выходом из приложения, чтобы корректно освободить ресурсы.

Kivy Основы

Простое приложение Kivy

from kivy.app import App
from kivy.uix.button import Button
from kivy.uix.boxlayout import BoxLayout
from kivy.uix.label import Label

class MyApp(App):
    def build(self):
        # Создание макета
        layout = BoxLayout(orientation='vertical', padding=20, spacing=10)
        
        # Создание виджетов
        label = Label(text='Привет, Kivy!', font_size='24sp')
        button = Button(text='Нажми меня', size_hint=(1, 0.3))
        button.bind(on_press=self.on_button_press)
        
        # Добавление виджетов в макет
        layout.add_widget(label)
        layout.add_widget(button)
        
        return layout
    
    def on_button_press(self, instance):
        print("Кнопка нажата!")
        instance.text = "Спасибо!"

# Запуск приложения
if __name__ == '__main__':
    MyApp().run()
Совет по Kivy

Используйте файлы .kv для разделения логики и интерфейса. Это делает код чище и упрощает поддержку.

ModernGL Основы

Минимальный пример ModernGL

import moderngl
import numpy as np
import pygame

# Инициализация Pygame и OpenGL контекста
pygame.init()
pygame.display.set_mode((800, 600), pygame.OPENGL | pygame.DOUBLEBUF)
ctx = moderngl.create_context()

# Шейдерная программа
prog = ctx.program(
    vertex_shader='''
        #version 330
        in vec2 in_vert;
        void main() {
            gl_Position = vec4(in_vert, 0.0, 1.0);
        }
    ''',
    fragment_shader='''
        #version 330
        out vec4 fragColor;
        void main() {
            fragColor = vec4(1.0, 0.5, 0.2, 1.0);
        }
    '''
)

# Данные вершин (треугольник)
vertices = np.array([
    -0.6, -0.6,
    0.6, -0.6,
    0.0, 0.6
], dtype='f4')

# Создание Vertex Buffer Object (VBO)
vbo = ctx.buffer(vertices.tobytes())
vao = ctx.simple_vertex_array(prog, vbo, 'in_vert')

# Главный цикл
running = True
while running:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
    
    # Очистка буфера цвета
    ctx.clear(0.1, 0.1, 0.1)
    
    # Рендеринг
    vao.render()
    
    # Обновление экрана
    pygame.display.flip()

pygame.quit()
Важно для ModernGL

ModernGL требует понимания основ OpenGL и графического конвейера. Начните с простых примеров перед созданием сложных сцен.

Советы по отладке PRO

  1. Используйте print() для быстрой отладки, но не забывайте удалять лишние выводы
  2. Используйте модуль logging для более структурированного логирования
  3. В PyGame используйте pygame.time.get_ticks() для отладки производительности
  4. В Kivy проверяйте журнал (log) для обнаружения ошибок в KV файлах
  5. В ModernGL убедитесь, что шейдеры компилируются без ошибок
